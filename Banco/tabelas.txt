CREATE TABLE routes(
 route_id int NOT NULL  AUTO_INCREMENT PRIMARY KEY,
 agency_id varchar (45),
 route_short_name varchar(100),
 route_long_name varchar (100),
 route_desc varchar(50),
route_type varchar(70),
route_url varchar(50),
route_color varchar(70),
route_text_color varchar (50)

);


CREATE TABLE usuario (
id_usuario int NOT NULL AUTO_INCREMENT,
nome_usuario varchar(100),
senha varchar (100)
);
